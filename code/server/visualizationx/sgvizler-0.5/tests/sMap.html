The element which draws the above chart:

<div id="sgvzl_example_query" 
   data-sgvizler-endpoint="http://sws.ifi.uio.no/sparql/enhetsregisteret" 
   data-sgvizler-query="PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> PREFIX dbpedia-owl: <http://dbpedia.org/ontology/> PREFIX dbpprop: <http://dbpedia.org/property/> PREFIX db: <http://dbpedia.org/resource/> PREFIX foaf: <http://xmlns.com/foaf/0.1/> PREFIX geo: <http://www.w3.org/2003/01/geo/wgs84_pos#> SELECT DISTINCT ?lat ?long ?name ?text ?url ?image WHERE { { ?url dbpedia-owl:country db:Germany . ?url foaf:name ?name . ?url geo:lat ?lat . ?url geo:long ?long . } UNION { ?url dbpedia-owl:country db:Germany . ?url dbpprop:name ?name . ?url geo:lat ?lat . ?url geo:long ?long . } }"
   data-sgvizler-chart="rdGraph" data-sgvizler-loglevel="2"
   style="width:1200px; height:500px;"/>
